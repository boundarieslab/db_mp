function getStrictLineage(startNode) {
let pathNodes = new Set();
let pathLinks = new Set();
pathNodes.add(startNode);
// Upwards
let queue = [startNode];
let visited = new Set([startNode.id]);
while(queue.length > 0) {
let curr = queue.shift();
// Find parents
let parents = links.filter(l => l.s === curr).map(l => l.t);
parents.forEach(p => {
let link = links.find(l => l.s === curr && l.t === p);
if(link) pathLinks.add(link);
pathNodes.add(p);
if(!visited.has(p.id)) { visited.add(p.id);
queue.push(p); }
});
}
// Downwards
function addChildren(pNode) {
let children = links.filter(l => l.t === pNode).map(l =>
l.s);
children.forEach(child => {
let link = links.find(l => l.s === child && l.t ===
pNode);
if(link) pathLinks.add(link);
pathNodes.add(child);
addChildren(child);
});
}
addChildren(startNode);
return {nodes: pathNodes, links: pathLinks};
}
